```typescript
const resend = new Resend(process.env.RESEND_API_KEY);
```

```typescript
// app/api/cron/email-sequence/route.ts
// รันทุกวัน เวลา 9:00 AM

export async function GET() {
  const supabase = createServerSupabase();

  // หา users ที่เข้าระบบมาแล้ว N วัน
  const sequences = [
    { days_after: 0, template: 'welcome' },
    { days_after: 1, template: 'pain-story' },
    { days_after: 3, template: 'social-proof' },
    { days_after: 5, template: 'educational' },
    { days_after: 7, template: 'feature-spotlight' },
    { days_after: 10, template: 'objection-handling' },
    { days_after: 14, template: 'last-chance' },
  ];

  for (const seq of sequences) {
    const targetDate = new Date();
    targetDate.setDate(targetDate.getDate() - seq.days_after);
    const dateStr = targetDate.toISOString().split('T')[0];

    const { data: leads } = await supabase
      .from('waitlist')
      .select('email, line_id')
      .gte('created_at', `${dateStr}T00:00:00`)
      .lt('created_at', `${dateStr}T23:59:59`);

    for (const lead of leads || []) {
      if (lead.email) {
        await sendEmail({ to: lead.email, template: seq.template });
      }
    }
  }

  return Response.json({ ok: true });
}
```

## ⚠️ Pitfalls ที่ต้องหลีกเลี่ยง

### ❌ ห้ามทำ:
1. ใช้ Subject Line คำว่า "Free", "Win", "Click here" → เข้า Spam
2. ใส่ลิงก์ > 3 อันต่ออีเมล → Spam Filter ไม่ชอบ
3. ส่งจากอีเมล Personal (gmail.com)
4. รูปภาพเยอะ → โหลดช้า + Spam
5. ส่งทุกวัน → Unsubscribe พุ่ง

### ✅ ต้องทำ:
1. ใช้ Domain ของตัวเอง (hello@alphascout.app)
2. ตั้งค่า SPF + DKIM + DMARC
3. Warm up Domain (ส่งจากน้อยๆ ก่อน)
4. ทดสอบ A/B testing subject lines
5. รับ Reply ทุกฉบับ (สร้าง Engagement)
